    风控模块

    max_position = 0.30        # 最大仓位30%
    stop_loss = 0.02          # 2%止损
    take_profit = 0.03       # 3%止盈
    
    # 杠杆参数
    max_leverage = 10        # 最大杠杆10x
    
    # 动态调整
    def calculate_position(self, volatility, trend):
        if volatility > 0.05:
